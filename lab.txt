//Selection sort
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

void swap(int *a, int *b) {
    int t = *a; *a = *b; *b = t;
}

void selectionSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int min = i;
        for (int j = i + 1; j < n; j++)
            if (arr[j] < arr[min]) min = j;
        swap(&arr[i], &arr[min]);
    }
}

int main() {
    int n = 10000;
    printf("Size,Time(ms)\n");

    for (int iter = 0; iter < 10; iter++) {
        int *a = malloc(n * sizeof(int));
        for (int i = 0; i < n; i++) a[i] = rand() % 100000;

        clock_t start = clock();
        selectionSort(a, n);
        clock_t end = clock();

        printf("%d,%ld\n", n, (end - start) * 1000 / CLOCKS_PER_SEC);
        free(a);
        n += 10000;
    }
    return 0;
}

//Quick Sort
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

void swap(int *a, int *b) { int t = *a; *a = *b; *b = t; }

int partition(int a[], int l, int h) {
    int p = a[h], i = l - 1;
    for (int j = l; j < h; j++)
        if (a[j] <= p) swap(&a[++i], &a[j]);
    swap(&a[i+1], &a[h]);
    return i + 1;
}

void quickSort(int a[], int l, int h) {
    if (l < h) {
        int pi = partition(a, l, h);
        quickSort(a, l, pi - 1);
        quickSort(a, pi + 1, h);
    }
}

int main() {
    int n; printf("Enter n: "); scanf("%d", &n);
    int a[n]; srand(time(0));
    for (int i = 0; i < n; i++) a[i] = rand() % 100;
    quickSort(a, 0, n - 1);
    for (int i = 0; i < n; i++) printf("%d ", a[i]);
}

//Merge sort
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

void merge(int a[], int l, int m, int r) {
    int n1 = m - l + 1, n2 = r - m;
    int *L = malloc(n1 * sizeof(int)), *R = malloc(n2 * sizeof(int));
    for (int i = 0; i < n1; i++) L[i] = a[l + i];
    for (int j = 0; j < n2; j++) R[j] = a[m + 1 + j];
    int i = 0, j = 0, k = l;
    while (i < n1 && j < n2) a[k++] = (L[i] <= R[j]) ? L[i++] : R[j++];
    while (i < n1) a[k++] = L[i++];
    while (j < n2) a[k++] = R[j++];
    free(L); free(R);
}

void mergeSort(int a[], int l, int r) {
    if (l >= r) return;
    int m = l + (r - l) / 2;
    mergeSort(a, l, m);
    mergeSort(a, m + 1, r);
    merge(a, l, m, r);
}

int main() {
    int n; printf("n: "); scanf("%d", &n);
    int *a = malloc(n * sizeof(int));
    srand(time(0));
    for (int i = 0; i < n; i++) a[i] = rand() % 100;
    mergeSort(a, 0, n - 1);
    for (int i = 0; i < n; i++) printf("%d ", a[i]);
    free(a);
}

//Kruskal
#include <stdio.h>
#define MAX 100

typedef struct {
    int u, v, cost;
} Edge;

int parent[MAX];

int find(int i) {
    while (parent[i]) i = parent[i];
    return i;
}

int unionSet(int i, int j) {
    int a = find(i), b = find(j);
    if (a != b) {
        parent[a] = b;
        return 1;
    }
    return 0;
}

int main() {
    int n, e, i, j, total = 0;
    Edge edges[MAX];
    
    printf("Vertices: ");
    scanf("%d", &n);
    printf("Edges: ");
    scanf("%d", &e);
    printf("Enter u v cost:\n");
    for (i = 0; i < e; i++)
        scanf("%d%d%d", &edges[i].u, &edges[i].v, &edges[i].cost);

    for (i = 0; i < e - 1; i++)
        for (j = 0; j < e - i - 1; j++)
            if (edges[j].cost > edges[j + 1].cost) {
                Edge tmp = edges[j];
                edges[j] = edges[j + 1];
                edges[j + 1] = tmp;
            }

    printf("MST Edges:\n");
    for (i = 0, j = 0; i < e && j < n - 1; i++)
        if (unionSet(edges[i].u, edges[i].v)) {
            printf("%d - %d : %d\n", edges[i].u, edges[i].v, edges[i].cost);
            total += edges[i].cost;
            j++;
        }

    printf("Total cost: %d\n", total);
    return 0;
}

//Prim's
#include <stdio.h>
#define MAX 100
#define INF 9999

int minKey(int key[], int mst[], int n) {
    int min = INF, idx = -1;
    for (int i = 0; i < n; i++)
        if (!mst[i] && key[i] < min)
            min = key[i], idx = i;
    return idx;
}

void printMST(int parent[], int g[MAX][MAX], int n) {
    int cost = 0;
    printf("Edge\tWeight\n");
    for (int i = 1; i < n; i++) {
        printf("%d - %d\t%d\n", parent[i], i, g[i][parent[i]]);
        cost += g[i][parent[i]];
    }
    printf("Total cost: %d\n", cost);
}

void primMST(int g[MAX][MAX], int n) {
    int parent[MAX], key[MAX], mst[MAX] = {0};
    for (int i = 0; i < n; i++) key[i] = INF;
    key[0] = 0, parent[0] = -1;

    for (int c = 0; c < n - 1; c++) {
        int u = minKey(key, mst, n);
        mst[u] = 1;
        for (int v = 0; v < n; v++)
            if (g[u][v] && !mst[v] && g[u][v] < key[v])
                parent[v] = u, key[v] = g[u][v];
    }
    printMST(parent, g, n);
}

int main() {
    int n, g[MAX][MAX];
    printf("Vertices: ");
    scanf("%d", &n);
    printf("Adjacency matrix:\n");
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++) {
            scanf("%d", &g[i][j]);
            if (!g[i][j]) g[i][j] = INF;
        }

    printf("\nMinimum Spanning Tree (Prim's):\n");
    primMST(g, n);
    return 0;
}
